170void HAL_GPIO_Init(GPIO_TypeDef *GPIOx, GPIO_InitTypeDef *GPIO_Init)
171{
172 uint32_t position = 0x00U;
173 uint32_t iocurrent;
174 uint32_t temp;
175
176 /* Check the parameters */
177 assert_param(IS_GPIO_ALL_INSTANCE(GPIOx));
178 assert_param(IS_GPIO_PIN(GPIO_Init->Pin));
179 assert_param(IS_GPIO_MODE(GPIO_Init->Mode));
180 assert_param(IS_GPIO_PULL(GPIO_Init->Pull));
181
182 /* Configure the port pins */
183 while (((GPIO_Init->Pin) >> position) != 0U)
184 {
185 /* Get current io position */
186 iocurrent = (GPIO_Init->Pin) & (1UL << position);
187
188 if (iocurrent != 0x00u)
189 {
190 /*--------------------- GPIO Mode Configuration ------------------------*/
191 /* In case of Output or Alternate function mode selection */
192 if ((GPIO_Init->Mode == GPIO_MODE_OUTPUT_PP) || (GPIO_Init->Mode == GPIO_MODE_AF_PP) ||
193 (GPIO_Init->Mode == GPIO_MODE_OUTPUT_OD) || (GPIO_Init->Mode == GPIO_MODE_AF_OD))
194 {
195 /* Check the Speed parameter */
196 assert_param(IS_GPIO_SPEED(GPIO_Init->Speed));
197 /* Configure the IO Speed */
198 temp = GPIOx->OSPEEDR;
199 temp &= ~(GPIO_OSPEEDR_OSPEED0 << (position * 2U));
200 temp |= (GPIO_Init->Speed << (position * 2U));
201 GPIOx->OSPEEDR = temp;
202
203 /* Configure the IO Output Type */
204 temp = GPIOx->OTYPER;
205 temp &= ~(GPIO_OTYPER_OT0 << position) ;
206 temp |= (((GPIO_Init->Mode & GPIO_OUTPUT_TYPE) >> 4U) << position);
207 GPIOx->OTYPER = temp;
208 }
209
210 /* Activate the Pull-up or Pull down resistor for the current IO */
211 temp = GPIOx->PUPDR;
212 temp &= ~(GPIO_PUPDR_PUPD0 << (position * 2U));
213 temp |= ((GPIO_Init->Pull) << (position * 2U));
214 GPIOx->PUPDR = temp;
215
216 /* In case of Alternate function mode selection */
217 if ((GPIO_Init->Mode == GPIO_MODE_AF_PP) || (GPIO_Init->Mode == GPIO_MODE_AF_OD))
218 {
219 /* Check the Alternate function parameters */
220 assert_param(IS_GPIO_AF_INSTANCE(GPIOx));
221 assert_param(IS_GPIO_AF(GPIO_Init->Alternate));
222
223 /* Configure Alternate function mapped with the current IO */
224 temp = GPIOx->AFR[position >> 3U];
225 temp &= ~(0xFU << ((position & 0x07U) * 4U));
226 temp |= ((GPIO_Init->Alternate) << ((position & 0x07U) * 4U));
227 GPIOx->AFR[position >> 3U] = temp;
228 }
229
230 /* Configure IO Direction mode (Input, Output, Alternate or Analog) */
231 temp = GPIOx->MODER;
232 temp &= ~(GPIO_MODER_MODE0 << (position * 2U));
233 temp |= ((GPIO_Init->Mode & GPIO_MODE) << (position * 2U));
234 GPIOx->MODER = temp;
235
236 /*--------------------- EXTI Mode Configuration ------------------------*/
237 /* Configure the External Interrupt or event for the current IO */
238 if ((GPIO_Init->Mode & EXTI_MODE) == EXTI_MODE)
239 {
240 /* Enable SYSCFG Clock */
241 __HAL_RCC_SYSCFG_CLK_ENABLE();
242
243 temp = SYSCFG->EXTICR[position >> 2U];
244 temp &= ~(0x0FUL << (4U * (position & 0x03U)));
245 temp |= (GPIO_GET_INDEX(GPIOx) << (4U * (position & 0x03U)));
246 SYSCFG->EXTICR[position >> 2U] = temp;
247
248 /* Clear EXTI line configuration */
249 temp = EXTI->IMR1;
250 temp &= ~(iocurrent);
251 if ((GPIO_Init->Mode & GPIO_MODE_IT) == GPIO_MODE_IT)
252 {
253 temp |= iocurrent;
254 }
255 EXTI->IMR1 = temp;
256
257 temp = EXTI->EMR1;
258 temp &= ~(iocurrent);
259 if ((GPIO_Init->Mode & GPIO_MODE_EVT) == GPIO_MODE_EVT)
260 {
261 temp |= iocurrent;
262 }
263 EXTI->EMR1 = temp;
264
265 /* Clear Rising Falling edge configuration */
266 temp = EXTI->RTSR1;
267 temp &= ~(iocurrent);
268 if ((GPIO_Init->Mode & RISING_EDGE) == RISING_EDGE)
269 {
270 temp |= iocurrent;
271 }
272 EXTI->RTSR1 = temp;
273
274 temp = EXTI->FTSR1;
275 temp &= ~(iocurrent);
276 if ((GPIO_Init->Mode & FALLING_EDGE) == FALLING_EDGE)
277 {
278 temp |= iocurrent;
279 }
280 EXTI->FTSR1 = temp;
281 }
282 }
283
284 position++;
285 }

295void HAL_GPIO_DeInit(GPIO_TypeDef *GPIOx, uint32_t GPIO_Pin)
296{
297 uint32_t position = 0x00U;
298 uint32_t iocurrent;
299 uint32_t tmp;
300
301 /* Check the parameters */
302 assert_param(IS_GPIO_ALL_INSTANCE(GPIOx));
303 assert_param(IS_GPIO_PIN(GPIO_Pin));
304
305 /* Configure the port pins */
306 while ((GPIO_Pin >> position) != 0U)
307 {
308 /* Get current io position */
309 iocurrent = (GPIO_Pin) & (1UL << position);
310
311 if (iocurrent != 0x00u)
312 {
313 /*------------------------- EXTI Mode Configuration --------------------*/
314 /* Clear the External Interrupt or Event for the current IO */
315
316 tmp = SYSCFG->EXTICR[position >> 2U];
317 tmp &= (0x0FUL << (4U * (position & 0x03U)));
318 if (tmp == (GPIO_GET_INDEX(GPIOx) << (4U * (position & 0x03U))))
319 {
320 /* Clear EXTI line configuration */
321 EXTI->IMR1 &= ~(iocurrent);
322 EXTI->EMR1 &= ~(iocurrent);
323
324 /* Clear Rising Falling edge configuration */
325 EXTI->RTSR1 &= ~(iocurrent);
326 EXTI->FTSR1 &= ~(iocurrent);
327
328 tmp = 0x0FUL << (4U * (position & 0x03U));
329 SYSCFG->EXTICR[position >> 2U] &= ~tmp;
330 }
331
332 /*------------------------- GPIO Mode Configuration --------------------*/
333 /* Configure IO in Analog Mode */
334 GPIOx->MODER |= (GPIO_MODER_MODE0 << (position * 2u));
335
336 /* Configure the default Alternate Function in current IO */
337 GPIOx->AFR[position >> 3u] &= ~(0xFu << ((position & 0x07u) * 4u));
338
339 /* Deactivate the Pull-up and Pull-down resistor for the current IO */
340 GPIOx->PUPDR &= ~(GPIO_PUPDR_PUPD0 << (position * 2u));
341
342 /* Configure the default value IO Output Type */
343 GPIOx->OTYPER &= ~(GPIO_OTYPER_OT0 << position);
344
345 /* Configure the default value for IO Speed */
346 GPIOx->OSPEEDR &= ~(GPIO_OSPEEDR_OSPEED0 << (position * 2u));
347 }
348
349 position++;
350 }

433void HAL_GPIO_TogglePin(GPIO_TypeDef *GPIOx, uint16_t GPIO_Pin)
434{
435 uint32_t odr;
436
437 /* Check the parameters */
438 assert_param(IS_GPIO_PIN(GPIO_Pin));
439
440 /* get current Output Data Register value */
441 odr = GPIOx->ODR;
442
443 /* Set selected pins that were at low level, and reset ones that were high */
444 GPIOx->BSRR = ((odr & GPIO_Pin) << GPIO_NUMBER) | (~odr & GPIO_Pin);
445}

458HAL_StatusTypeDef HAL_GPIO_LockPin(GPIO_TypeDef *GPIOx, uint16_t GPIO_Pin)
459{
460 __IO uint32_t tmp = GPIO_LCKR_LCKK;
461
462 /* Check the parameters */
463 assert_param(IS_GPIO_LOCK_INSTANCE(GPIOx));
464 assert_param(IS_GPIO_PIN(GPIO_Pin));
465
466 /* Apply lock key write sequence */
467 tmp |= GPIO_Pin;
468 /* Set LCKx bit(s): LCKK='1' + LCK[15-0] */
469 GPIOx->LCKR = tmp;
470 /* Reset LCKx bit(s): LCKK='0' + LCK[15-0] */
471 GPIOx->LCKR = GPIO_Pin;
472 /* Set LCKx bit(s): LCKK='1' + LCK[15-0] */
473 GPIOx->LCKR = tmp;
474 /* Read LCKK register. This read is mandatory to complete key lock sequence */
475 tmp = GPIOx->LCKR;
476
477 /* read again in order to confirm lock is active */
478 if ((GPIOx->LCKR & GPIO_LCKR_LCKK) != 0x00u)
479 {
480 return HAL_OK;
481 }
482 else
483 {
484 return HAL_ERROR;
485 }
