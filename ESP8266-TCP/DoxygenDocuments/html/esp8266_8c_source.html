205Status Command_Process(char **commandArray, char **responseArray, uint8_t numberOfCommands)
206{
207 static uint8_t commandFlag = 1, currentCommand = 0;
208 Status response;
209
210 if(!numberOfCommands || currentCommand >= numberOfCommands)
211 {
212 ringBuffer_flush(rx_buffer);
213 return STATUS_ERROR;
214 }
215 else
216 {
217 if(commandFlag)
218 {
219 Send_AT_Command(commandArray[currentCommand]);
220 commandFlag = 0;
221 }
222 response = Wait_Response(responseArray[currentCommand], 5000);
223
224 if(Read_Response("ERROR"))
225 {
226 commandFlag = 1;
227 currentCommand = 0;
228 ringBuffer_flush(rx_buffer);
229 return STATUS_ERROR;
230 }
231 else if(response == IDLE)
232 {
233 return response;
234 }
235 else if(response == FOUND)
236 {
237 if(currentCommand == (numberOfCommands-1))
238 {
239 commandFlag = 1;
240 currentCommand = 0;
241 ringBuffer_flush(rx_buffer);
242 return STATUS_OK;
243 }
244 else
245 {
246 commandFlag = 1;
247 currentCommand += 1;
248 ringBuffer_flush(rx_buffer);
249 return response;
250 }
251
252 }
253 else
254 {
255 commandFlag = 1;
256 currentCommand = 0;
257 return response;
258 }
