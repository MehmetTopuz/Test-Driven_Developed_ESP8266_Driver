140{
141 HAL_StatusTypeDef status;
142 uint32_t page_index;
143
144 /* Check the parameters */
145 assert_param(IS_FLASH_TYPEERASE(pEraseInit->TypeErase));
146
147 /* Process Locked */
148 __HAL_LOCK(&pFlash);
149
150 /* Wait for last operation to be completed */
151 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
152
153 if (status == HAL_OK)
154 {
155 pFlash.ErrorCode = HAL_FLASH_ERROR_NONE;
156
157 /* Deactivate the cache if they are activated to avoid data misbehavior */
158 if (READ_BIT(FLASH->ACR, FLASH_ACR_ICEN) != 0U)
159 {
160 /* Disable instruction cache */
161 __HAL_FLASH_INSTRUCTION_CACHE_DISABLE();
162
163 if (READ_BIT(FLASH->ACR, FLASH_ACR_DCEN) != 0U)
164 {
165 /* Disable data cache */
166 __HAL_FLASH_DATA_CACHE_DISABLE();
167 pFlash.CacheToReactivate = FLASH_CACHE_ICACHE_DCACHE_ENABLED;
168 }
169 else
170 {
171 pFlash.CacheToReactivate = FLASH_CACHE_ICACHE_ENABLED;
172 }
173 }
174 else if (READ_BIT(FLASH->ACR, FLASH_ACR_DCEN) != 0U)
175 {
176 /* Disable data cache */
177 __HAL_FLASH_DATA_CACHE_DISABLE();
178 pFlash.CacheToReactivate = FLASH_CACHE_DCACHE_ENABLED;
179 }
180 else
181 {
182 pFlash.CacheToReactivate = FLASH_CACHE_DISABLED;
183 }
184
185 if (pEraseInit->TypeErase == FLASH_TYPEERASE_MASSERASE)
186 {
187 /* Mass erase to be done */
188 FLASH_MassErase(pEraseInit->Banks);
189
190 /* Wait for last operation to be completed */
191 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
192
193#if defined (FLASH_OPTR_DBANK)
194 /* If the erase operation is completed, disable the MER1 and MER2 Bits */
195 CLEAR_BIT(FLASH->CR, (FLASH_CR_MER1 | FLASH_CR_MER2));
196#else
197 /* If the erase operation is completed, disable the MER1 Bit */
198 CLEAR_BIT(FLASH->CR, (FLASH_CR_MER1));
199#endif
200 }
201 else
202 {
203 /*Initialization of PageError variable*/
204 *PageError = 0xFFFFFFFFU;
205
206 for (page_index = pEraseInit->Page; page_index < (pEraseInit->Page + pEraseInit->NbPages); page_index++)
207 {
208 FLASH_PageErase(page_index, pEraseInit->Banks);
209
210 /* Wait for last operation to be completed */
211 status = FLASH_WaitForLastOperation((uint32_t)FLASH_TIMEOUT_VALUE);
212
213 /* If the erase operation is completed, disable the PER Bit */
214 CLEAR_BIT(FLASH->CR, (FLASH_CR_PER | FLASH_CR_PNB));
215
216 if (status != HAL_OK)
217 {
218 /* In case of error, stop erase procedure and return the faulty page */
219 *PageError = page_index;
220 break;
221 }
222 }
223 }
224
225 /* Flush the caches to be sure of the data consistency */
226 FLASH_FlushCaches();
227 }
228
229 /* Process Unlocked */
230 __HAL_UNLOCK(&pFlash);
231
232 return status;

613 SET_BIT(FLASH->CR, FLASH_CR_STRT);

624 /* Flush instruction cache */

626 (cache == FLASH_CACHE_ICACHE_DCACHE_ENABLED))

629 __HAL_FLASH_INSTRUCTION_CACHE_RESET();
630 /* Enable instruction cache */
631 __HAL_FLASH_INSTRUCTION_CACHE_ENABLE();
632 }
633
634 /* Flush data cache */
635 if ((cache == FLASH_CACHE_DCACHE_ENABLED) ||
636 (cache == FLASH_CACHE_ICACHE_DCACHE_ENABLED))
637 {
638 /* Reset data cache */
639 __HAL_FLASH_DATA_CACHE_RESET();
640 /* Enable data cache */
641 __HAL_FLASH_DATA_CACHE_ENABLE();
642 }
643
644 /* Reset internal variable */
645 pFlash.CacheToReactivate = FLASH_CACHE_DISABLED;
646}
