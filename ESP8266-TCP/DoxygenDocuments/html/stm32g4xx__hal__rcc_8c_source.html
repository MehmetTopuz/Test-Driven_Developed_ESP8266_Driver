217HAL_StatusTypeDef HAL_RCC_DeInit(void)
218{
219 uint32_t tickstart;
220
221 /* Get Start Tick*/
222 tickstart = HAL_GetTick();
223
224 /* Set HSION bit to the reset value */
225 SET_BIT(RCC->CR, RCC_CR_HSION);
226
227 /* Wait till HSI is ready */
228 while (READ_BIT(RCC->CR, RCC_CR_HSIRDY) == 0U)
229 {
230 if ((HAL_GetTick() - tickstart) > HSI_TIMEOUT_VALUE)
231 {
232 return HAL_TIMEOUT;
233 }
234 }
235
236 /* Set HSITRIM[6:0] bits to the reset value */
237 SET_BIT(RCC->ICSCR, RCC_HSICALIBRATION_DEFAULT << RCC_ICSCR_HSITRIM_Pos);
238
239 /* Get Start Tick*/
240 tickstart = HAL_GetTick();
241
242 /* Reset CFGR register (HSI is selected as system clock source) */
243 RCC->CFGR = 0x00000001u;
244
245 /* Wait till HSI is ready */
246 while (READ_BIT(RCC->CFGR, RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI)
247 {
248 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
249 {
250 return HAL_TIMEOUT;
251 }
252 }
253
254 /* Update the SystemCoreClock global variable */
255 SystemCoreClock = HSI_VALUE;
256
257 /* Adapt Systick interrupt period */
258 if (HAL_InitTick(uwTickPrio) != HAL_OK)
259 {
260 return HAL_ERROR;
261 }
262
263 /* Clear CR register in 2 steps: first to clear HSEON in case bypass was enabled */
264 RCC->CR = RCC_CR_HSION;
265
266 /* Then again to HSEBYP in case bypass was enabled */
267 RCC->CR = RCC_CR_HSION;
268
269 /* Get Start Tick*/
270 tickstart = HAL_GetTick();
271
272 /* Wait till PLL is OFF */
273 while (READ_BIT(RCC->CR, RCC_CR_PLLRDY) != 0U)
274 {
275 if ((HAL_GetTick() - tickstart) > PLL_TIMEOUT_VALUE)
276 {
277 return HAL_TIMEOUT;
278 }
279 }
280
281 /* once PLL is OFF, reset PLLCFGR register to default value */
282 RCC->PLLCFGR = RCC_PLLCFGR_PLLN_4;
283
284 /* Disable all interrupts */
285 CLEAR_REG(RCC->CIER);
286
287 /* Clear all interrupt flags */
288 WRITE_REG(RCC->CICR, 0xFFFFFFFFU);
289
290 /* Clear all reset flags */
291 SET_BIT(RCC->CSR, RCC_CSR_RMVF);
292
293 return HAL_OK;
294}

765HAL_StatusTypeDef HAL_RCC_ClockConfig(RCC_ClkInitTypeDef *RCC_ClkInitStruct, uint32_t FLatency)
766{
767 uint32_t tickstart;
768 uint32_t pllfreq;
769 uint32_t hpre = RCC_SYSCLK_DIV1;
770
771 /* Check Null pointer */
772 if (RCC_ClkInitStruct == NULL)
773 {
774 return HAL_ERROR;
775 }
776
777 /* Check the parameters */
778 assert_param(IS_RCC_CLOCKTYPE(RCC_ClkInitStruct->ClockType));
779 assert_param(IS_FLASH_LATENCY(FLatency));
780
781 /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
782 must be correctly programmed according to the frequency of the CPU clock
783 (HCLK) and the supply voltage of the device. */
784
785 /* Increasing the number of wait states because of higher CPU frequency */
786 if (FLatency > __HAL_FLASH_GET_LATENCY())
787 {
788 /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
789 __HAL_FLASH_SET_LATENCY(FLatency);
790
791 /* Check that the new number of wait states is taken into account to access the Flash
792 memory by reading the FLASH_ACR register */
793 if (__HAL_FLASH_GET_LATENCY() != FLatency)
794 {
795 return HAL_ERROR;
796 }
797 }
798
799 /*------------------------- SYSCLK Configuration ---------------------------*/
800 if(((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_SYSCLK) == RCC_CLOCKTYPE_SYSCLK)
801 {
802 assert_param(IS_RCC_SYSCLKSOURCE(RCC_ClkInitStruct->SYSCLKSource));
803
804 /* PLL is selected as System Clock Source */
805 if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_PLLCLK)
806 {
807 /* Check the PLL ready flag */
808 if (READ_BIT(RCC->CR, RCC_CR_PLLRDY) == 0U)
809 {
810 return HAL_ERROR;
811 }
812 /* Undershoot management when selection PLL as SYSCLK source and frequency above 80Mhz */
813 /* Compute target PLL output frequency */
814 pllfreq = RCC_GetSysClockFreqFromPLLSource();
815
816 /* Intermediate step with HCLK prescaler 2 necessary before to go over 80Mhz */
817 if(pllfreq > 80000000U)
818 {
819 if (((READ_BIT(RCC->CFGR, RCC_CFGR_HPRE) == RCC_SYSCLK_DIV1)) ||
820 (((((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_HCLK) == RCC_CLOCKTYPE_HCLK) &&
821 (RCC_ClkInitStruct->AHBCLKDivider == RCC_SYSCLK_DIV1))))
822 {
823 MODIFY_REG(RCC->CFGR, RCC_CFGR_HPRE, RCC_SYSCLK_DIV2);
824 hpre = RCC_SYSCLK_DIV2;
825 }
826 }
827 }
828 else
829 {
830 /* HSE is selected as System Clock Source */
831 if (RCC_ClkInitStruct->SYSCLKSource == RCC_SYSCLKSOURCE_HSE)
832 {
833 /* Check the HSE ready flag */
834 if(READ_BIT(RCC->CR, RCC_CR_HSERDY) == 0U)
835 {
836 return HAL_ERROR;
837 }
838 }
839 /* HSI is selected as System Clock Source */
840 else
841 {
842 /* Check the HSI ready flag */
843 if(READ_BIT(RCC->CR, RCC_CR_HSIRDY) == 0U)
844 {
845 return HAL_ERROR;
846 }
847 }
848 /* Overshoot management when going down from PLL as SYSCLK source and frequency above 80Mhz */
849 pllfreq = HAL_RCC_GetSysClockFreq();
850
851 /* Intermediate step with HCLK prescaler 2 necessary before to go under 80Mhz */
852 if(pllfreq > 80000000U)
853 {
854 MODIFY_REG(RCC->CFGR, RCC_CFGR_HPRE, RCC_SYSCLK_DIV2);
855 hpre = RCC_SYSCLK_DIV2;
856 }
857
858 }
859
860 MODIFY_REG(RCC->CFGR, RCC_CFGR_SW, RCC_ClkInitStruct->SYSCLKSource);
861
862 /* Get Start Tick*/
863 tickstart = HAL_GetTick();
864
865 while (__HAL_RCC_GET_SYSCLK_SOURCE() != (RCC_ClkInitStruct->SYSCLKSource << RCC_CFGR_SWS_Pos))
866 {
867 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
868 {
869 return HAL_TIMEOUT;
870 }
871 }
872 }
873
874 /*-------------------------- HCLK Configuration --------------------------*/
875 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_HCLK) == RCC_CLOCKTYPE_HCLK)
876 {
877 /* Set the highest APB divider in order to ensure that we do not go through
878 a non-spec phase whatever we decrease or increase HCLK. */
879 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK1) == RCC_CLOCKTYPE_PCLK1)
880 {
881 MODIFY_REG(RCC->CFGR, RCC_CFGR_PPRE1, RCC_HCLK_DIV16);
882 }
883 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK2) == RCC_CLOCKTYPE_PCLK2)
884 {
885 MODIFY_REG(RCC->CFGR, RCC_CFGR_PPRE2, RCC_HCLK_DIV16);
886 }
887
888 /* Set the new HCLK clock divider */
889 assert_param(IS_RCC_HCLK(RCC_ClkInitStruct->AHBCLKDivider));
890 MODIFY_REG(RCC->CFGR, RCC_CFGR_HPRE, RCC_ClkInitStruct->AHBCLKDivider);
891 }
892 else
893 {
894 /* Is intermediate HCLK prescaler 2 applied internally, complete with HCLK prescaler 1 */
895 if(hpre == RCC_SYSCLK_DIV2)
896 {
897 MODIFY_REG(RCC->CFGR, RCC_CFGR_HPRE, RCC_SYSCLK_DIV1);
898 }
899 }
900
901 /* Decreasing the number of wait states because of lower CPU frequency */
902 if (FLatency < __HAL_FLASH_GET_LATENCY())
903 {
904 /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
905 __HAL_FLASH_SET_LATENCY(FLatency);
906
907 /* Check that the new number of wait states is taken into account to access the Flash
908 memory by polling the FLASH_ACR register */
909 tickstart = HAL_GetTick();
910
911 while (__HAL_FLASH_GET_LATENCY() != FLatency)
912 {
913 if ((HAL_GetTick() - tickstart) > CLOCKSWITCH_TIMEOUT_VALUE)
914 {
915 return HAL_TIMEOUT;
916 }
917 }
918 }
919
920 /*-------------------------- PCLK1 Configuration ---------------------------*/
921 if (((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK1) == RCC_CLOCKTYPE_PCLK1)
922 {
923 assert_param(IS_RCC_PCLK(RCC_ClkInitStruct->APB1CLKDivider));
924 MODIFY_REG(RCC->CFGR, RCC_CFGR_PPRE1, RCC_ClkInitStruct->APB1CLKDivider);
925 }
926
927 /*-------------------------- PCLK2 Configuration ---------------------------*/
928 if(((RCC_ClkInitStruct->ClockType) & RCC_CLOCKTYPE_PCLK2) == RCC_CLOCKTYPE_PCLK2)

1046{

1048 uint32_t sysclockfreq;

1051 {
1052 /* HSI used as system clock source */
1053 sysclockfreq = HSI_VALUE;
1054 }
1055 else if (__HAL_RCC_GET_SYSCLK_SOURCE() == RCC_CFGR_SWS_HSE)
1056 {
1057 /* HSE used as system clock source */

1059 }

1065 SYSCLK = PLL_VCO / PLLR
1066 */

1068 pllm = (READ_BIT(RCC->PLLCFGR, RCC_PLLCFGR_PLLM) >> RCC_PLLCFGR_PLLM_Pos) + 1U ;
1069

1071 {
1072 case RCC_PLLSOURCE_HSE: /* HSE used as PLL clock source */
1073 pllvco = (HSE_VALUE / pllm) * (READ_BIT(RCC->PLLCFGR, RCC_PLLCFGR_PLLN) >> RCC_PLLCFGR_PLLN_Pos);
1074 break;
1075
1076 case RCC_PLLSOURCE_HSI: /* HSI used as PLL clock source */
1077 default:
1078 pllvco = (HSI_VALUE / pllm) * (READ_BIT(RCC->PLLCFGR, RCC_PLLCFGR_PLLN) >> RCC_PLLCFGR_PLLN_Pos);
1079 break;
1080 }

1082 sysclockfreq = pllvco/pllr;
1083 }
1084 else
1085 {
1086 sysclockfreq = 0U;
1087 }
