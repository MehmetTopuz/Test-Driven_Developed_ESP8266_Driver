309HAL_StatusTypeDef HAL_PWR_ConfigPVD(PWR_PVDTypeDef *sConfigPVD)
310{
311 /* Check the parameters */
312 assert_param(IS_PWR_PVD_LEVEL(sConfigPVD->PVDLevel));
313 assert_param(IS_PWR_PVD_MODE(sConfigPVD->Mode));
314
315 /* Set PLS bits according to PVDLevel value */
316 MODIFY_REG(PWR->CR2, PWR_CR2_PLS, sConfigPVD->PVDLevel);
317
318 /* Clear any previous config. Keep it clear if no event or IT mode is selected */
319 __HAL_PWR_PVD_EXTI_DISABLE_EVENT();
320 __HAL_PWR_PVD_EXTI_DISABLE_IT();
321 __HAL_PWR_PVD_EXTI_DISABLE_FALLING_EDGE();
322 __HAL_PWR_PVD_EXTI_DISABLE_RISING_EDGE();
323
324 /* Configure interrupt mode */
325 if((sConfigPVD->Mode & PVD_MODE_IT) == PVD_MODE_IT)
326 {
327 __HAL_PWR_PVD_EXTI_ENABLE_IT();
328 }
329
330 /* Configure event mode */
331 if((sConfigPVD->Mode & PVD_MODE_EVT) == PVD_MODE_EVT)
332 {
333 __HAL_PWR_PVD_EXTI_ENABLE_EVENT();
334 }
335
336 /* Configure the edge */
337 if((sConfigPVD->Mode & PVD_RISING_EDGE) == PVD_RISING_EDGE)
338 {
339 __HAL_PWR_PVD_EXTI_ENABLE_RISING_EDGE();
340 }
341
342 if((sConfigPVD->Mode & PVD_FALLING_EDGE) == PVD_FALLING_EDGE)
343 {
344 __HAL_PWR_PVD_EXTI_ENABLE_FALLING_EDGE();
345 }
346
347 return HAL_OK;
348}

442void HAL_PWR_EnterSLEEPMode(uint32_t Regulator, uint8_t SLEEPEntry)
443{
444 /* Check the parameters */
445 assert_param(IS_PWR_REGULATOR(Regulator));
446 assert_param(IS_PWR_SLEEP_ENTRY(SLEEPEntry));
447
448 /* Set Regulator parameter */
449 if (Regulator == PWR_MAINREGULATOR_ON)
450 {
451 /* If in low-power run mode at this point, exit it */
452 if (HAL_IS_BIT_SET(PWR->SR2, PWR_SR2_REGLPF))
453 {
454 (void)HAL_PWREx_DisableLowPowerRunMode();
455 }
456 /* Regulator now in main mode. */
457 }
458 else
459 {
460 /* If in run mode, first move to low-power run mode.
461 The system clock frequency must be below 2 MHz at this point. */
462 if (HAL_IS_BIT_SET(PWR->SR2, PWR_SR2_REGLPF) == 0U)
463 {
464 HAL_PWREx_EnableLowPowerRunMode();
465 }
466 }
467
468 /* Clear SLEEPDEEP bit of Cortex System Control Register */
469 CLEAR_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
470
471 /* Select SLEEP mode entry -------------------------------------------------*/
472 if(SLEEPEntry == PWR_SLEEPENTRY_WFI)
473 {
474 /* Request Wait For Interrupt */
475 __WFI();
476 }
477 else
478 {
479 /* Request Wait For Event */
480 __SEV();
481 __WFE();
482 __WFE();
483 }
484
485}

550void HAL_PWR_EnterSTANDBYMode(void)
551{
552 /* Set Stand-by mode */
553 MODIFY_REG(PWR->CR1, PWR_CR1_LPMS, PWR_CR1_LPMS_STANDBY);
554
555 /* Set SLEEPDEEP bit of Cortex System Control Register */
556 SET_BIT(SCB->SCR, ((uint32_t)SCB_SCR_SLEEPDEEP_Msk));
557
558/* This option is used to ensure that store operations are completed */
559#if defined ( __CC_ARM)
560 __force_stores();
561#endif
562 /* Request Wait For Interrupt */
563 __WFI();
564}
